\subsection{Important Definitions}
\begin{definition}[Continuity]
    A function $f: \R \to \R$ is \emph{continuous} at $x_0 \in \R$ if
    \[
    \forall \epsilon > 0, \exists \delta > 0 : |x - x_0| < \delta \implies |f(x) - f(x_0)| < \epsilon.
    \]
\end{definition}
